339  void handle_swap_entity(entity a, entity b)
340  {
341  using std::swap;
342  auto c_a = components_.find(a);
343  auto c_b = components_.find(b);
344  auto end = components_.end();
345 
346  if(c_a != end && c_b != end)
347  {
348  swap(c_a->second, c_b->second);
349  }
350  else if(c_a != end)
351  {
352  components_.emplace(b, std::move(c_a->second));
353  components_.erase(a);
354  }
355  else if(c_b != end)
356  {
357  components_.emplace(a, std::move(c_b->second));
358  components_.erase(b);
359  }
360  }
